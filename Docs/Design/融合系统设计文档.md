# 宠物收集塔防游戏《Pet Collector》融合系统设计文档

## 1. 系统概述

融合系统是《Pet Collector》游戏的核心玩法之一，允许玩家将进化到最终等级的宠物与特定机器人进行融合，创造出更强大的终极单位。融合后的单位可以装备六件装备，大幅提升战斗力。

## 2. 系统架构

### 2.1 FusionSystem (融合系统)
- **职责**: 处理各类融合逻辑，管理融合配方，计算融合结果
- **类型**: 静态系统类
- **依赖**: MythicalPetData, UnitData, PlayerData

```csharp
/// <summary>
/// 尝试融合两个对象
/// </summary>
public static FusionResult FuseObjects(object obj1, object obj2)
{
    // 检查输入参数
    if (obj1 == null || obj2 == null)
    {
        return new FusionResult(false, null, "融合对象不能为空！");
    }

    // 根据对象类型执行相应的融合逻辑
    if (obj1 is MythicalPetData && obj2 is UnitData)
    {
        return FusePetWithRobot((MythicalPetData)obj1, (UnitData)obj2);
    }
    else if (obj1 is UnitData && obj2 is MythicalPetData)
    {
        return FusePetWithRobot((MythicalPetData)obj2, (UnitData)obj1);
    }
    else if (obj1 is UnitData && obj2 is UnitData)
    {
        return FuseUnits((UnitData)obj1, (UnitData)obj2);
    }
    else if (obj1 is MythicalPetData && obj2 is MythicalPetData)
    {
        return FusePets((MythicalPetData)obj1, (MythicalPetData)obj2);
    }
    else
    {
        return new FusionResult(false, null, "不支持的融合组合！");
    }
}
```

#### 3.4.2 宠物与机器人融合方法
```csharp
/// <summary>
/// 宠物与机器人融合
/// </summary>
public static FusionResult FusePetWithRobot(MythicalPetData pet, UnitData robot)
{
    // 检查融合条件
    if (pet == null || robot == null)
    {
        return new FusionResult(false, null, "融合对象不能为空！");
    }

    // 检查宠物是否达到最终进化等级
    if (pet.EvolutionLevel < 5)
    {
        return new FusionResult(false, null, "宠物需要进化到第5级才能融合！");
    }

    // 检查机器人是否达到最高等级
    if (robot.Level < 10)
    {
        return new FusionResult(false, null, "机器人需要达到10级才能融合！");
    }

    // 检查融合兼容性
    if (pet.FusionCompatibility < 0.5f)
    {
        return new FusionResult(false, null, "宠物融合兼容性不足！");
    }

    // 检查玩家是否有足够资源
    if (GameManager.Instance?.PlayerData.Souls < 50)
    {
        return new FusionResult(false, null, "灵魂不足，需要50点灵魂进行融合！");
    }

    // 检查机器人是否可融合
    if (!robot.CanFuse)
    {
        return new FusionResult(false, null, "该机器人不可用于融合！");
    }

    // 创建融合后的单位
    UnitData fusedUnit = new UnitData();
    
    // 设置基本信息
    fusedUnit.Id = UnityEngine.Random.Range(10000, 99999);
    fusedUnit.Name = $"{pet.Name}-{robot.Name} Fusion";
    fusedUnit.Description = $"由{pet.Name}和{robot.Name}融合而成的强大单位";
    fusedUnit.Type = UnitData.UnitType.PetHybrid;
    
    // 合并属性（宠物属性和机器人属性的加权合并）
    fusedUnit.AttackPower = (pet.AttackPower * 0.6f + robot.AttackPower * 0.4f) * 1.3f; // 30%融合加成
    fusedUnit.DefensePower = (pet.DefensePower * 0.5f + robot.DefensePower * 0.5f) * 1.3f;
    fusedUnit.Speed = (pet.Speed * 0.4f + robot.Speed * 0.6f) * 1.2f; // 机器人对速度贡献更大
    fusedUnit.Health = (pet.Health * 0.5f + robot.Health * 0.5f) * 1.4f; // 40%融合加成
    fusedUnit.Range = Mathf.Max(pet.SkillRange, robot.Range) * 1.1f; // 取较大值并略微提升
    
    // 保留宠物的特殊能力
    fusedUnit.AttackPower += pet.DivinePower * 0.5f;
    fusedUnit.DefensePower += pet.ProtectionPower * 0.5f;
    
    // 设置融合后的特殊属性
    fusedUnit.CanFuse = false; // 融合后的单位不能再融合
    fusedUnit.IsFinalForm = true; // 达到最终形态
    fusedUnit.MaxEquipmentSlots = FusionConfig.PET_ROBOT_FUSION_EQUIPMENT_SLOTS; // 6个装备槽位
    fusedUnit.Level = Mathf.Max(pet.Level, robot.Level); // 取较高等级
    fusedUnit.Experience = (pet.Experience + robot.Experience) / 2; // 平均经验
    
    // 消耗资源
    if (GameManager.Instance?.PlayerData != null)
    {
        GameManager.Instance.PlayerData.AddSouls(-50);
        GameManager.Instance.PlayerData.AddGems(-10);
    }

    string message = $"融合成功！{pet.Name}和{robot.Name}融合成为强大的{fusedUnit.Name}！拥有{fusedUnit.MaxEquipmentSlots}个装备槽位！";

    return new FusionResult(true, fusedUnit, message, fusedUnit.MaxEquipmentSlots);
}
```

```csharp
/// <summary>
/// 机器人与机器人融合
/// </summary>
public static FusionResult FuseUnits(UnitData unit1, UnitData unit2)
{
    // 检查融合条件
    if (unit1 == null || unit2 == null)
    {
        return new FusionResult(false, null, "融合对象不能为空！");
    }

    // 检查两个单位是否都可以融合
    if (!unit1.CanFuse || !unit2.CanFuse)
    {
        return new FusionResult(false, null, "其中一个单位不可融合！");
    }

    // 检查等级要求
    if (unit1.Level < 8 || unit2.Level < 8)
    {
        return new FusionResult(false, null, "两个机器人都需要达到8级才能融合！");
    }

    // 检查玩家是否有足够资源
    if (GameManager.Instance?.PlayerData.Coins < 500 || GameManager.Instance?.PlayerData.Gems < 20)
    {
        return new FusionResult(false, null, "资源不足，需要500金币和20神石进行融合！");
    }

    // 创建融合后的单位
    UnitData fusedUnit = new UnitData();
    
    // 设置基本信息
    fusedUnit.Id = UnityEngine.Random.Range(10000, 99999);
    fusedUnit.Name = $"{unit1.Name}-{unit2.Name} Synthesis";
    fusedUnit.Description = $"由{unit1.Name}和{unit2.Name}合成的高级单位";
    fusedUnit.Type = UnitData.UnitType.Hybrid;
    
    // 合并属性
    fusedUnit.AttackPower = (unit1.AttackPower + unit2.AttackPower) * 1.25f; // 25%合成加成
    fusedUnit.DefensePower = (unit1.DefensePower + unit2.DefensePower) * 1.25f;
    fusedUnit.Speed = (unit1.Speed + unit2.Speed) / 2 * 1.1f; // 10%速度加成
    fusedUnit.Health = (unit1.Health + unit2.Health) * 1.3f; // 30%血量加成
    fusedUnit.Range = Mathf.Max(unit1.Range, unit2.Range) * 1.15f; // 15%射程加成
    
    // 保留较高的研发等级
    fusedUnit.RequiredResearchLevel = Mathf.Max(unit1.RequiredResearchLevel, unit2.RequiredResearchLevel);
    
    // 设置融合后的特殊属性
    fusedUnit.CanFuse = true; // 仍可进一步融合
    fusedUnit.IsFinalForm = false; // 不是最终形态
    fusedUnit.MaxEquipmentSlots = FusionConfig.ROBOT_ROBOT_FUSION_EQUIPMENT_SLOTS; // 4个装备槽位
    fusedUnit.Level = Mathf.Max(unit1.Level, unit2.Level) + 1; // 等级提升
    fusedUnit.Experience = (unit1.Experience + unit2.Experience) / 2;
    
    // 消耗资源
    if (GameManager.Instance?.PlayerData != null)
    {
        GameManager.Instance.PlayerData.AddCoins(-500);
        GameManager.Instance.PlayerData.AddGems(-20);
    }

    string message = $"合成成功！{unit1.Name}和{unit2.Name}合成为{fusedUnit.Name}！拥有{fusedUnit.MaxEquipmentSlots}个装备槽位！";

    return new FusionResult(true, fusedUnit, message, fusedUnit.MaxEquipmentSlots);
}
```

```csharp
/// <summary>
/// 宠物与宠物融合
/// </summary>
public static FusionResult FusePets(MythicalPetData pet1, MythicalPetData pet2)
{
    // 检查融合条件
    if (pet1 == null || pet2 == null)
    {
        return new FusionResult(false, null, "融合对象不能为空！");
    }

    // 检查两个宠物是否都可以融合
    if (pet1.EvolutionLevel < 3 || pet2.EvolutionLevel < 3)
    {
        return new FusionResult(false, null, "两个宠物都需要进化到第3级才能融合！");
    }

    // 检查玩家是否有足够资源
    if (GameManager.Instance?.PlayerData.Souls < 30)
    {
        return new FusionResult(false, null, "灵魂不足，需要30点灵魂进行融合！");
    }

    // 创建融合后的宠物
    MythicalPetData fusedPet = new MythicalPetData();
    
    // 设置基本信息
    fusedPet.Id = UnityEngine.Random.Range(1000, 9999);
    fusedPet.Name = $"{pet1.Name}+{pet2.Name}";
    fusedPet.Description = $"由{pet1.Name}和{pet2.Name}融合而成的新宠物";
    fusedPet.Element = DetermineCombinedElement(pet1.Element, pet2.Element);
    fusedPet.MythologicalBackground = $"{pet1.MythologicalBackground}与{pet2.MythologicalBackground}的结合";
    
    // 合并属性
    fusedPet.AttackPower = (pet1.AttackPower + pet2.AttackPower) * 1.2f; // 20%融合加成
    fusedPet.DefensePower = (pet1.DefensePower + pet2.DefensePower) * 1.2f;
    fusedPet.Speed = (pet1.Speed + pet2.Speed) / 2 * 1.15f; // 15%速度加成
    fusedPet.Health = (pet1.Health + pet2.Health) * 1.25f; // 25%血量加成
    
    // 提升稀有度（如果两个都是高级稀有度）
    if ((int)pet1.Rarity >= 3 && (int)pet2.Rarity >= 3) // DivineBeast及以上
    {
        fusedPet.Rarity = (MythicalPetData.MythicalRarity)Mathf.Max((int)pet1.Rarity, (int)pet2.Rarity);
        if ((int)fusedPet.Rarity < 4) // 如果还不是最高级，则提升一级
        {
            fusedPet.Rarity = (MythicalPetData.MythicalRarity)((int)fusedPet.Rarity + 1);
        }
    }
    else
    {
        fusedPet.Rarity = (MythicalPetData.MythicalRarity)Mathf.Max((int)pet1.Rarity, (int)pet2.Rarity);
    }
    
    // 神话属性融合
    fusedPet.DivinePower = (pet1.DivinePower + pet2.DivinePower) * 1.1f;
    fusedPet.ProtectionPower = (pet1.ProtectionPower + pet2.ProtectionPower) * 1.1f;
    fusedPet.PurificationPower = (pet1.PurificationPower + pet2.PurificationPower) * 1.1f;
    
    // 设置融合后的特殊属性
    fusedPet.EvolutionLevel = Mathf.Max(pet1.EvolutionLevel, pet2.EvolutionLevel) + 1; // 进化等级提升
    fusedPet.EvolutionProgress = (pet1.EvolutionProgress + pet2.EvolutionProgress) / 2;
    fusedPet.EvolutionStonesRequired = Mathf.Max(pet1.EvolutionStonesRequired, pet2.EvolutionStonesRequired) + 15;
    fusedPet.CanFuseWithRobots = fusedPet.EvolutionLevel >= 5; // 根据进化等级确定是否可融合
    fusedPet.FusionCompatibility = (pet1.FusionCompatibility + pet2.FusionCompatibility) / 2 * 1.1f; // 兼容性提升
    
    // 消耗资源
    if (GameManager.Instance?.PlayerData != null)
    {
        GameManager.Instance.PlayerData.AddSouls(-30);
    }

    string message = $"融合成功！{pet1.Name}和{pet2.Name}融合成为{fusedPet.Name}！稀有度提升至{fusedPet.Rarity}！";

    return new FusionResult(true, fusedPet, message, 0); // 宠物融合不产生装备槽位
}
```

#### 3.4.5 辅助方法
```csharp
/// <summary>
/// 确定融合后的元素属性
/// </summary>
private static string DetermineCombinedElement(string element1, string element2)
{
    // 简单的元素融合规则
    if (element1 == element2) return element1; // 相同元素保持不变
    
    // 不同元素的融合规则
    string[] elements = { "Fire", "Water", "Wood", "Metal", "Earth" };
    
    // 如果两个元素相邻，则可能产生新元素
    for (int i = 0; i < elements.Length; i++)
    {
        if ((elements[i] == element1 && elements[(i + 1) % elements.Length] == element2) ||
            (elements[i] == element2 && elements[(i + 1) % elements.Length] == element1))
        {
            // 例如：火+土=熔岩，水+气=雾等
            if ((element1 == "Fire" && element2 == "Earth") || (element1 == "Earth" && element2 == "Fire"))
                return "Lava";
            else if ((element1 == "Water" && element2 == "Air") || (element1 == "Air" && element2 == "Water"))
                return "Vapor";
            else if ((element1 == "Fire" && element2 == "Metal") || (element1 == "Metal" && element2 == "Fire"))
                return "Plasma";
        }
    }
    
    // 默认返回第一个元素
    return element1;
}

/// <summary>
/// 检查是否可以融合
/// </summary>
public static bool CanFuse(object obj1, object obj2)
{
    if (obj1 == null || obj2 == null) return false;

    // 根据对象类型检查融合条件
    if (obj1 is MythicalPetData && obj2 is UnitData)
    {
        return CanPetFuseWithRobot((MythicalPetData)obj1, (UnitData)obj2);
    }
    else if (obj1 is UnitData && obj2 is MythicalPetData)
    {
        return CanPetFuseWithRobot((MythicalPetData)obj2, (UnitData)obj1);
    }
    else if (obj1 is UnitData && obj2 is UnitData)
    {
        return CanUnitFuseWithUnit((UnitData)obj1, (UnitData)obj2);
    }
    else if (obj1 is MythicalPetData && obj2 is MythicalPetData)
    {
        return CanPetFuseWithPet((MythicalPetData)obj1, (MythicalPetData)obj2);
    }
    
    return false;
}

/// <summary>
/// 检查宠物是否可以与机器人融合
/// </summary>
public static bool CanPetFuseWithRobot(MythicalPetData pet, UnitData robot)
{
    if (pet == null || robot == null) return false;
    
    // 检查宠物进化等级
    if (pet.EvolutionLevel < 5) return false;
    
    // 检查机器人等级
    if (robot.Level < 10) return false;
    
    // 检查融合兼容性
    if (pet.FusionCompatibility < 0.5f) return false;
    
    // 检查机器人是否可融合
    if (!robot.CanFuse) return false;
    
    // 检查玩家资源
    if (GameManager.Instance?.PlayerData.Souls < 50) return false;
    
    return true;
}
```

## 4. 系统交互

### 4.1 数据流
```
PlayerData → FusionSystem → Fused UnitData/MythicalPetData
UI → FusionSystem → Fusion Result
EquipmentSystem → FusionSystem → Equipment Slots Info
```

### 4.2 事件流
- 玩家发起融合请求 → FusionSystem 验证条件 → 执行融合 → 返回结果 → UI 更新显示
- 融合成功 → PlayerData 更新单位列表 → EquipmentSystem 启用装备功能

### 4.3 依赖关系
- FusionSystem 依赖 PlayerData 验证资源
- FusionSystem 依赖 MythicalPetData 和 UnitData 获取对象属性
- EquipmentSystem 依赖 FusionSystem 获取装备槽位信息
